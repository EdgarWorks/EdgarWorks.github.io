Text Cleaning — CRIME_DATA_CLEAN_TEXT

Internal spacing issue: some LOCATION values contain multiple consecutive spaces from the original paper-transcribed data. A CHAR(7) sentinel character technique is used to collapse them without corrupting valid single spaces.
Step 05 — Text Normalization Rules Applied CLEAN 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 05: TEXT CLEANING — Normalize all string columns
--
--  Rules applied:
--   1. TRIM()        → Remove leading/trailing whitespace
--   2. UPPER()       → Standardize to uppercase
--   3. NULLIF(,'')   → Convert empty strings to NULL
--   4. CHAR(7) trick → Collapse internal multiple spaces to single space
--      Pattern: '  ' → ' '+CHAR(7) → CHAR(7)+' '→ '' → remove CHAR(7)
-- ═══════════════════════════════════════════════════════════════════

SELECT
    NULLIF(TRIM(DR_NO), '')                                               AS DR_NO,
    NULLIF(TRIM(Date_Rptd), '')                                           AS DATE_RPTD,
    NULLIF(TRIM(DATE_OCC), '')                                            AS DATE_OCC,
    NULLIF(TRIM(TIME_OCC), '')                                            AS TIME_OCC,
    NULLIF(TRIM(AREA), '')                                                AS AREA,
    NULLIF(UPPER(TRIM(AREA_NAME)), '')                                    AS AREA_NAME,
    NULLIF(TRIM(Rpt_Dist_No), '')                                         AS DISTRICT,
    NULLIF(TRIM(Part_1_2), '')                                            AS PART1_2,
    NULLIF(TRIM(Crm_Cd), '')                                              AS CRM_CD,
    NULLIF(UPPER(TRIM(Crm_Cd_Desc)), '')                                  AS CRM_CD_DESC,
    NULLIF(TRIM(Mocodes), '')                                             AS MOCODES,
    NULLIF(TRIM(Vict_Age), '')                                            AS VICT_AGE,
    NULLIF(UPPER(TRIM(Vict_Sex)), '')                                     AS VICT_SEX,
    NULLIF(UPPER(TRIM(Vict_Descent)), '')                                AS VICT_DESCENT,
    NULLIF(TRIM(Premis_Cd), '')                                           AS PREMIS_CD,
    NULLIF(UPPER(TRIM(Premis_Desc)), '')                                  AS PREMIS_DESC,
    NULLIF(TRIM(Weapon_Used_Cd), '')                                      AS WEAPON_CD,
    NULLIF(UPPER(TRIM(Weapon_Desc)), '')                                  AS WEAPON_DESC,
    NULLIF(UPPER(TRIM(Status)), '')                                       AS STATUS_ID,
    NULLIF(UPPER(TRIM(Status_Desc)), '')                                  AS STATUS_DESC,
    NULLIF(TRIM(Crm_Cd_1), '')                                            AS CRM_CD_1,
    NULLIF(TRIM(Crm_Cd_2), '')                                            AS CRM_CD_2,
    NULLIF(TRIM(Crm_Cd_3), '')                                            AS CRM_CD_3,
    NULLIF(TRIM(Crm_Cd_4), '')                                            AS CRM_CD_4,
    -- Collapse internal multiple spaces using CHAR(7) sentinel technique
    NULLIF(UPPER(TRIM(
        REPLACE(REPLACE(REPLACE(LOCATION,
            '  ', ' ' + CHAR(7)),
            CHAR(7) + ' ', ''),
            CHAR(7), '')
    )), '')                                                                  AS LOCATION,
    NULLIF(UPPER(TRIM(
        REPLACE(REPLACE(REPLACE(Cross_Street,
            '  ', ' ' + CHAR(7)),
            CHAR(7) + ' ', ''),
            CHAR(7), '')
    )), '')                                                                  AS CROSS_STREET,
    NULLIF(TRIM(LAT), '')                                                AS LATITUDE,
    NULLIF(TRIM(LON), '')                                                AS LONGITUDE
INTO CRIME_DATA_CLEAN_TEXT
FROM CRIME_DATA_RAW_WORK;
GO

DROP TABLE CRIME_DATA_RAW_WORK; -- No longer needed

Duplicate Detection & Removal

2,966 duplicate rows detected and removed. DR_NO is unique (zero key duplicates). Duplicates identified across all 27 non-key columns using ROW_NUMBER() OVER PARTITION BY.
Step 06 — Duplicate Isolation using Window Functions ROW_NUMBER() 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 06: DUPLICATE DETECTION
--
--  Strategy: ROW_NUMBER() OVER PARTITION BY all non-key columns.
--   cnt_reg = 1  → unique record → keep
--   cnt_reg > 1  → duplicate     → archive to DUPLICATES table
--
--  Note: DR_NO column excluded from partition — it is always unique.
--  Some incidents share identical data with different DR_NOs (data
--  entry duplicates from paper-to-digital transcription).
-- ═══════════════════════════════════════════════════════════════════

WITH REG_DUP AS (
    SELECT *,
        COUNT(*) OVER (
            PARTITION BY
                Date_Rptd, DATE_OCC, TIME_OCC,
                AREA, AREA_NAME, DISTRICT,
                PART1_2, Crm_Cd, Crm_Cd_Desc, Mocodes,
                Vict_Age, Vict_Sex, Vict_Descent,
                Premis_Cd, Premis_Desc, WEAPON_CD, Weapon_Desc,
                STATUS_ID, Status_Desc,
                Crm_Cd_1, Crm_Cd_2, Crm_Cd_3, Crm_Cd_4,
                LOCATION, Cross_Street, LATITUDE, LONGITUDE
        ) AS total_duplicates,
        ROW_NUMBER() OVER (
            PARTITION BY
                Date_Rptd, DATE_OCC, TIME_OCC,
                AREA, AREA_NAME, DISTRICT,
                PART1_2, Crm_Cd, Crm_Cd_Desc, Mocodes,
                Vict_Age, Vict_Sex, Vict_Descent,
                Premis_Cd, Premis_Desc, WEAPON_CD, Weapon_Desc,
                STATUS_ID, Status_Desc,
                Crm_Cd_1, Crm_Cd_2, Crm_Cd_3, Crm_Cd_4,
                LOCATION, Cross_Street, LATITUDE, LONGITUDE
            ORDER BY DR_NO                       -- Lowest DR_NO = canonical record
        ) AS row_rank
    FROM CRIME_DATA_CLEAN_TEXT
)
-- Archive duplicates for audit
SELECT * INTO CRIME_DATA_DUPLICATES
FROM REG_DUP WHERE row_rank > 1;
GO

-- Build clean table (row_rank = 1 only)
WITH REG_DUP AS ( /* same CTE as above */ )
SELECT
    DR_NO, Date_Rptd, DATE_OCC, TIME_OCC,
    AREA, AREA_NAME, DISTRICT, PART1_2,
    Crm_Cd, Crm_Cd_Desc, Mocodes,
    Vict_Age, Vict_Sex, Vict_Descent,
    Premis_Cd, Premis_Desc, WEAPON_CD, Weapon_Desc,
    STATUS_ID, Status_Desc,
    Crm_Cd_1, Crm_Cd_2, Crm_Cd_3, Crm_Cd_4,
    LOCATION, Cross_Street, LATITUDE, LONGITUDE
INTO CRIME_DATA_CLEAN_TEXT_WITHOUT_DUPLICATES
FROM REG_DUP WHERE row_rank = 1;
GO
-- Result: 1,002,096 rows (1,005,062 − 2,966 duplicates)

Missing Values Analysis — Stored Procedure

A reusable stored procedure scans every column of any table using dynamic SQL, calculating completeness %, missing count, and a recommended action (keep/delete/impute/drop column).
sp_ResumenValoresFaltantesTabla — Dynamic SQL Table Scanner STORED PROC 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 07: REUSABLE STORED PROCEDURE FOR MISSING VALUE ANALYSIS
--
--  Input:  @la_tabla  (table name), @el_esquema (schema, default 'dbo')
--  Output: Per-column completeness report with action recommendation
--
--  Recommendation thresholds:
--   = 0%    → KEEP INTACT
--   ≤ 10%   → DELETE NULL ROWS
--   ≤ 60%   → IMPUTE (Mean / Mode)
--   ≤ 90%   → EVALUATE RELEVANCE
--   > 90%   → DROP COLUMN (low informativeness)
-- ═══════════════════════════════════════════════════════════════════

CREATE PROCEDURE sp_ResumenValoresFaltantesTabla
    @la_tabla   NVARCHAR(128),
    @el_esquema NVARCHAR(128) = 'dbo'
AS
BEGIN
    SET NOCOUNT ON;

    DECLARE @sql                   NVARCHAR(MAX) = N'';
    DECLARE @nombre_completo_tabla  NVARCHAR(256) =
        QUOTENAME(@el_esquema) + '.' + QUOTENAME(@la_tabla);

    -- Build one SELECT per column, union-all them
    SELECT @sql = @sql +
        'SELECT ' +
            '''' + name + ''' AS nombre_columna, ' +
            'COUNT(*) AS tot_de_registros, ' +
            'SUM(CASE WHEN ' + QUOTENAME(name) +
            ' IS NULL OR LTRIM(RTRIM(CAST(' + QUOTENAME(name) +
            ' AS NVARCHAR(MAX)))) = '''' THEN 1 ELSE 0 END) AS tot_reg_faltantes ' +
        'FROM ' + @nombre_completo_tabla + ' UNION ALL '
    FROM sys.columns
    WHERE object_id = OBJECT_ID(@nombre_completo_tabla);

    -- Remove trailing UNION ALL
    SET @sql = LEFT(@sql, LEN(@sql) - 10);

    -- Wrap in final recommendation logic
    DECLARE @final_sql NVARCHAR(MAX) = N'
    SELECT
        nombre_columna,
        tot_de_registros,
        tot_reg_faltantes,
        TRY_CONVERT(DECIMAL(6,2),
            100.0*(tot_de_registros-tot_reg_faltantes)/NULLIF(tot_de_registros,0)
        ) AS x100_completitud,
        TRY_CONVERT(DECIMAL(6,2),
            100.0*tot_reg_faltantes/NULLIF(tot_de_registros,0)
        ) AS x100_faltantes,
        CASE
            WHEN (100.0*tot_reg_faltantes/NULLIF(tot_de_registros,0)) =   0
                THEN ''KEEP INTACT''
            WHEN (100.0*tot_reg_faltantes/NULLIF(tot_de_registros,0)) <=  10
                THEN ''DELETE NULL ROWS''
            WHEN (100.0*tot_reg_faltantes/NULLIF(tot_de_registros,0)) <=  60
                THEN ''IMPUTE (Mean/Mode)''
            WHEN (100.0*tot_reg_faltantes/NULLIF(tot_de_registros,0)) <=  90
                THEN ''EVALUATE RELEVANCE''
            ELSE     ''DROP COLUMN (low informativeness)''
        END AS RECOMMENDATION
    FROM (' + @sql + ') T1
    ORDER BY x100_faltantes DESC';

    EXEC sp_executesql @final_sql;
END
GO

Time Format Correction — HHMM → HH:MM:00 UPDATE 
-- ═══════════════════════════════════════════════════════════════════
-- TIME_OCC: Convert 4-digit integer format to HH:MM:00
--
--  Example: '2130' → '21:30:00' (21:30 PM)
--           '0845' → '08:45:00'  (8:45 AM)
--           '0'    → '00:00:00'  (midnight)
--
--  Logic: RIGHT('0000'+value, 4) → zero-pad to 4 digits
--         LEFT(padded, 2)        → extract hours
--         RIGHT(padded, 2)       → extract minutes
-- ═══════════════════════════════════════════════════════════════════

UPDATE CRIME_DATA_CLEAN_TEXT_WITHOUT_DUPLICATES
SET    TIME_OCC =
    LEFT(RIGHT('0000' + TIME_OCC, 4), 2) + ':' +
    RIGHT(RIGHT('0000' + TIME_OCC, 4), 2) + ':00'
WHERE  DR_NO IS NOT NULL;
GO

UPDATE CRIME_DATA_CLEAN_TEXT_WITHOUT_DUPLICATES

Outlier Analysis — Victim Age (Tukey IQR Method)

VICTIM_AGE is the only truly quantitative column in this dataset. All other numeric columns are codes/IDs/coordinates that don't support classical outlier analysis. Tukey IQR method applied (Q3 + 1.5×IQR).
Step 09 — Complete Statistical Profile with Percentiles PERCENTILE_CONT 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 09: VICTIM AGE STATISTICAL ANALYSIS
--
--  Note: PERCENTILE_CONT cannot be mixed with aggregate functions
--  (COUNT, AVG, etc.) in the same SELECT. Use separate CTEs.
--
--  Results:
--   Total w/age: 733,026 | Min: 2 | Max: 120 | Mean: 39.51
--   Q1: 28 | Median: 37 | Q3: 50 | IQR: 22
--   Tukey upper fence: 83 → outliers start at 84
--   Outliers (Tukey): 4,364 records (0.60%)
--   Outliers (Z-score ±3σ): 2,641 records (0.36%)
-- ═══════════════════════════════════════════════════════════════════

WITH AGE AS (
    SELECT victim_age
    FROM   CRIME_DATA_TYPED
    WHERE  victim_age IS NOT NULL
),
BASIC_STATS AS (
    SELECT
        COUNT(*)                                        AS total_reg,
        COUNT(DISTINCT victim_age)                     AS unique_age,
        MIN(victim_age)                                 AS minimum,
        MAX(victim_age)                                 AS maximum,
        CAST(AVG(CAST(victim_age AS FLOAT)) AS DECIMAL(10,2)) AS mean,
        CAST(VAR(victim_age)  AS DECIMAL(10,2))         AS variance,
        CAST(STDEV(victim_age) AS DECIMAL(10,2))         AS stdev,
        MAX(victim_age) - MIN(victim_age)              AS range_val
    FROM AGE
),
QUARTILES AS (
    -- PERCENTILE_CONT must be isolated from aggregates
    SELECT DISTINCT
        CAST(PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY victim_age) OVER () AS DECIMAL(10,2)) AS q1,
        CAST(PERCENTILE_CONT(0.50) WITHIN GROUP (ORDER BY victim_age) OVER () AS DECIMAL(10,2)) AS q2_median,
        CAST(PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY victim_age) OVER () AS DECIMAL(10,2)) AS q3,
        CAST(PERCENTILE_CONT(0.99) WITHIN GROUP (ORDER BY victim_age) OVER () AS DECIMAL(10,2)) AS p99
    FROM AGE
)
SELECT
    B.*,
    Q.q1, Q.q2_median, Q.q3,
    CAST(Q.q3 - Q.q1 AS DECIMAL(10,2))                          AS iqr,
    CAST(Q.q1 - 1.5 * (Q.q3 - Q.q1) AS DECIMAL(10,2))            AS tukey_lower_fence,
    CAST(Q.q3 + 1.5 * (Q.q3 - Q.q1) AS DECIMAL(10,2))            AS tukey_upper_fence,   -- = 83
    CAST(B.mean - 3 * B.stdev AS DECIMAL(10,2))                  AS zscore_lower_fence,
    CAST(B.mean + 3 * B.stdev AS DECIMAL(10,2))                  AS zscore_upper_fence,
    Q.p99
FROM BASIC_STATS B
CROSS JOIN QUARTILES Q;
GO

-- Decision: Ages 84–99 are biologically valid (high-frequency, declining pattern).
--           Age 120 (1 record): victim_sex=NULL, victim_descent=NULL → likely data
--           entry error. All age > 83 archived to CRIME_DATA_OUTLIERS_VICTIM_AGE.
--           CRIME_DATA_TYPED remains complete (source of truth).
--           CRIME_DATA_ANALYSIS view excludes outliers for statistical work.

SELECT * INTO CRIME_DATA_OUTLIERS_VICTIM_AGE
FROM CRIME_DATA_TYPED
WHERE victim_age > 83;
GO

Typed Table — CRIME_DATA_TYPED

Final clean table with proper SQL Server data types, expanded categorical codes to full text, age = 0 converted to NULL, derived age classification columns added.
Step 10 — SELECT INTO with Full Type Casting CAST / CONVERT 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 10: CREATE CRIME_DATA_TYPED — Final typed clean table
--
--  Key decisions:
--   · victim_age = 0   → NULL  (no victim present, not a real age)
--   · VICT_SEX codes   → Expanded: M→MALE, F→FEMALE, X→UNKNOWN
--   · VICT_DESCENT     → Expanded to full ethnicity names (20 codes)
--   · DATE columns     → CONVERT(DATE, col, 101) removes time portion
--   · TIME_OCC         → CAST AS TIME(0) — no fractional seconds
--   · All ELSE → NULL  (no 'NOT APPLICABLE' catch-all strings)
-- ═══════════════════════════════════════════════════════════════════

SELECT
    -- ── IDENTIFICATION ─────────────────────────────────────────
    CAST(DR_NO        AS BIGINT)                    AS report_number,

    -- ── DATES & TIME ────────────────────────────────────────────
    TRY_CONVERT(DATE, DATE_OCC,  101)                  AS date_occurred,
    TRY_CONVERT(DATE, DATE_RPTD, 101)                  AS date_reported,
    CAST(TIME_OCC    AS TIME(0))                     AS time_occurred,

    -- ── AREA & DISTRICT ─────────────────────────────────────────
    CAST(AREA        AS TINYINT)                     AS area,
    CAST(AREA_NAME   AS NVARCHAR(50))                 AS area_name,
    CAST(DISTRICT   AS SMALLINT)                    AS district,
    CAST(PART1_2    AS TINYINT)                     AS class_crime_type,

    -- ── CRIME CODE ──────────────────────────────────────────────
    CAST(CRM_CD      AS SMALLINT)                    AS crime_code,
    CAST(CRM_CD_DESC AS NVARCHAR(100))                AS crime_code_description,
    CAST(MOCODES     AS NVARCHAR(100))                AS modus_operandus,

    -- ── VICTIM ──────────────────────────────────────────────────
    CASE
        WHEN CAST(VICT_AGE AS TINYINT) = 0 THEN NULL  -- 0 = no victim
        ELSE                              CAST(VICT_AGE AS TINYINT)
    END                                                   AS victim_age,

    CAST(CASE
        WHEN VICT_SEX = 'M' THEN 'MALE'
        WHEN VICT_SEX = 'F' THEN 'FEMALE'
        WHEN VICT_SEX = 'X' THEN 'UNKNOWN'
        ELSE                     NULL    -- NULL = no victim present
    END AS NVARCHAR(15))                                  AS victim_sex,

    CAST(CASE
        WHEN VICT_DESCENT = 'A' THEN 'ASIAN'
        WHEN VICT_DESCENT = 'B' THEN 'BLACK'
        WHEN VICT_DESCENT = 'C' THEN 'CHINESE'
        WHEN VICT_DESCENT = 'D' THEN 'CAMBODIAN'
        WHEN VICT_DESCENT = 'F' THEN 'PHILIPPINE'
        WHEN VICT_DESCENT = 'G' THEN 'GUAMANIAN'
        WHEN VICT_DESCENT = 'H' THEN 'HISPANIC'
        WHEN VICT_DESCENT = 'I' THEN 'AMERICAN INDIAN'
        WHEN VICT_DESCENT = 'J' THEN 'JAPANESE'
        WHEN VICT_DESCENT = 'K' THEN 'KOREAN'
        WHEN VICT_DESCENT = 'L' THEN 'LAOTIAN'
        WHEN VICT_DESCENT = 'O' THEN 'OTHER'
        WHEN VICT_DESCENT = 'P' THEN 'PACIFIC ISLANDER'
        WHEN VICT_DESCENT = 'S' THEN 'SAMOAN'
        WHEN VICT_DESCENT = 'U' THEN 'HAWAIIAN'
        WHEN VICT_DESCENT = 'V' THEN 'VIETNAMESE'
        WHEN VICT_DESCENT = 'W' THEN 'WHITE'
        WHEN VICT_DESCENT = 'X' THEN 'UNKNOWN'
        WHEN VICT_DESCENT = 'Z' THEN 'ASIAN INDIAN'
        ELSE                          NULL
    END AS NVARCHAR(30))                                  AS victim_descent,

    -- ── PREMISE ─────────────────────────────────────────────────
    CAST(PREMIS_CD    AS SMALLINT)                    AS premis_code,
    CAST(PREMIS_DESC  AS NVARCHAR(100))                AS premis_description,

    -- ── WEAPON (NULL = no weapon involved) ──────────────────────
    NULLIF(TRY_CAST(WEAPON_CD   AS SMALLINT), '')       AS weapon_used_code,
    NULLIF(TRY_CAST(WEAPON_DESC AS NVARCHAR(100)), '') AS weapon_description,

    -- ── CASE STATUS ─────────────────────────────────────────────
    CAST(STATUS_ID    AS CHAR(2))                      AS status,
    CAST(STATUS_DESC  AS NVARCHAR(20))                  AS status_description,

    -- ── ADDITIONAL CHARGES (multivalued → normalized in Step 11) ─
    TRY_CAST(CRM_CD_1 AS SMALLINT)                     AS crime_code_1,
    TRY_CAST(CRM_CD_2 AS SMALLINT)                     AS crime_code_2,
    TRY_CAST(CRM_CD_3 AS SMALLINT)                     AS crime_code_3,
    TRY_CAST(CRM_CD_4 AS SMALLINT)                     AS crime_code_4,

    -- ── LOCATION ────────────────────────────────────────────────
    CAST(LOCATION     AS NVARCHAR(60))                  AS location_occurred,
    NULLIF(TRY_CAST(CROSS_STREET AS NVARCHAR(60)), '') AS cross_street,
    CAST(LATITUDE    AS DECIMAL(8,4))                  AS latitude,
    CAST(LONGITUDE   AS DECIMAL(8,4))                  AS longitude

INTO CRIME_DATA_TYPED
FROM  CRIME_DATA_CLEAN_TEXT_WITHOUT_DUPLICATES;
GO
-- Result: 999,299 rows — source of truth for all downstream objects
// 11

Normalization — Multivalued Columns

Two columns contain multiple values per row: MOCODES (space-separated 4-digit codes) and CRM_CD_1–4 (additional crime charges). Both are denormalized into junction tables using STRING_SPLIT() and CROSS APPLY with composite primary keys.
Step 11 — Junction Tables: RECORDS_AND_MOCODES & RECORDS_AND_CRIMECODES STRING_SPLIT 
-- ═══════════════════════════════════════════════════════════════════
-- STEP 11A: NORMALIZE MODUS OPERANDUS CODES
--           Each space-separated 4-digit mocode becomes one row.
--           Composite PK prevents any report_number + mocode duplicate.
-- ═══════════════════════════════════════════════════════════════════

CREATE TABLE RECORDS_AND_MOCODES (
    report_number   BIGINT    NOT NULL,
    modus_operandus SMALLINT  NOT NULL,
    PRIMARY KEY (report_number, modus_operandus)
);
GO

INSERT INTO RECORDS_AND_MOCODES
SELECT DISTINCT
    report_number,
    CAST(value AS SMALLINT) AS modus_operandus
FROM CRIME_DATA_TYPED
CROSS APPLY STRING_SPLIT(modus_operandus, ' ')
WHERE value <> ''
  AND value IS NOT NULL
  AND TRY_CAST(value AS SMALLINT) IS NOT NULL;
GO
-- Result: 2,947,966 rows (avg ~2.95 mocodes per report)

-- ═══════════════════════════════════════════════════════════════════
-- STEP 11B: NORMALIZE CRIME CODES (CRM_CD + CRM_CD_1 through CRM_CD_4)
--           CONCAT_WS combines all 5 code columns into one string,
--           then STRING_SPLIT explodes each code to a separate row.
--           Code 998 (filler placeholder) is excluded.
-- ═══════════════════════════════════════════════════════════════════

CREATE TABLE RECORDS_AND_CRIMECODES (
    record_number   BIGINT    NOT NULL,
    crime_code      SMALLINT  NOT NULL,
    PRIMARY KEY (record_number, crime_code)
);
GO

INSERT INTO RECORDS_AND_CRIMECODES
SELECT DISTINCT
    report_number,
    TRY_CAST(value AS SMALLINT) AS crime_code
FROM CRIME_DATA_TYPED
CROSS APPLY STRING_SPLIT(
    CONCAT_WS(' ',
        CAST(crime_code   AS NVARCHAR(10)),
        CAST(crime_code_1 AS NVARCHAR(10)),
        CAST(crime_code_2 AS NVARCHAR(10)),
        CAST(crime_code_3 AS NVARCHAR(10)),
        CAST(crime_code_4 AS NVARCHAR(10))
    ), ' ')
WHERE value <> ''
  AND value <> '998'     -- Exclude filler code
  AND TRY_CAST(value AS SMALLINT) IS NOT NULL;
GO
-- Result: 1,070,466 rows

Analytical Views — Architecture

Design principle: Never JOIN two multivalued tables in a permanent view. Doing so creates a Cartesian product (N×M rows per report) that inflates row counts dramatically. Each view joins at most ONE multivalued table.
CRIME_DATA_PRINCIPAL
999,299 rows — enriched with date parts, age groups, crime flags
vSUMMARY_REPORTS
999,299 rows
One row per report. Crime code joined directly from CRIME_DATA_TYPED. No multivalued joins. Fastest view — dashboards & KPIs.
vSUMMARY_REPORTS_WITH_CRIME
1,070,466 rows
One row per report × crime code. Joins RECORDS_AND_CRIMECODES + catalog. Use for crime category analysis.
vSUMMARY_REPORTS_WITH_MOCODES
2,947,966 rows
One row per report × mocode. Joins RECORDS_AND_MOCODES + catalog. Use for modus operandi analysis.
vCODES_CATALOG
3,025,453 rows
Codes only — no report data. Cross-joins both multivalued tables. Always filter by crime_code or mocode_category before querying.
CRIME_DATA_ANALYSIS (VIEW)
~994,935 rows
Filter view over CRIME_DATA_TYPED. Excludes victim_age > 83 (Tukey outliers). Use for statistical analysis.
vSUMMARY_REPORTS — 1:1 Master View (KPIs & Dashboards) VIEW 
-- ═══════════════════════════════════════════════════════════════════
-- VIEW: vSUMMARY_REPORTS — One row per incident report
--
--  Joins CRIME_DATA_PRINCIPAL + CRIME_DATA_TYPED (for primary crime
--  code) + CRIME_CODE_DESC_CLASS (for crime category).
--  No multivalued table joins → always returns exactly 999,299 rows.
--  Use this view for 90% of general analysis and dashboards.
-- ═══════════════════════════════════════════════════════════════════

CREATE VIEW vSUMMARY_REPORTS AS
SELECT
    P.report_number,
    P.date_occurred,        P.year_occurred,
    UPPER(P.month_occurred)  AS month_occurred,
    P.day_of_week_occurred,   P.quarter_of_year_occurred,
    P.time_occurred,         P.hour_occurred,
    P.period_of_day,          P.lapd_shift,
    P.is_daytime,             P.is_nighttime,       P.is_business_hours,
    P.date_reported,          P.day_for_reported,
    P.reporting_immediacy,    P.report_temperature,
    P.is_late_report,         P.is_very_late_report,
    -- Primary crime code — from typed table (no junction join)
    T.crime_code,
    T.crime_code_description,
    CC.crime_category,
    P.area,                   P.area_name,           P.district,
    P.class_crime_type,        P.classification_crime_name,
    P.victim_age,              P.age_group_10yr,       P.life_stage,
    P.legal_status,            P.lapd_age_category,   P.generation,
    P.vulnerability_group,     P.broad_age_group,
    P.victim_sex,              P.victim_descent,
    P.premis_code,             P.premis_description,
    NULLIF(P.weapon_used_code,  '') AS weapon_used_code,
    NULLIF(P.weapon_description, '') AS weapon_description,
    P.status,                  P.status_description,
    P.location_occurred,
    NULLIF(P.cross_street, '')  AS cross_street,
    P.latitude,               P.longitude
FROM  CRIME_DATA_PRINCIPAL    P
INNER JOIN CRIME_DATA_TYPED   T  ON P.report_number = T.report_number
LEFT JOIN  CRIME_CODE_DESC_CLASS CC ON T.crime_code    = CC.crime_code;
GO
-- Verification: SELECT COUNT(*) FROM vSUMMARY_REPORTS → 999,299
